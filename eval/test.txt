Private Function JoinRange(arr() As String, ByVal i1 As Long, ByVal i2 As Long) As String
    Dim i As Long, t As String
    For i = i1 To i2
        If Len(arr(i)) > 0 Then
            If Len(t) > 0 Then t = t & vbCrLf & arr(i) Else t = arr(i)
        End If
    Next
    JoinRange = t
End Function
